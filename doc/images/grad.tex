\begin{align*}
  \nabla F &= grad*F \\
  F \nabla &= F*grad \\
  \nabla \wedge F &= grad \wedge F \\
  F \wedge \nabla &= F \wedge grad \\
  \nabla \cdot F &= grad|F \\
  F \cdot \nabla F &= F|grad \\
  \nabla \lfloor F &= grad  <  F \\
  F \lfloor \nabla &= F  <  grad \\
  \nabla \rfloor F &= grad  >  F \\
  F \rfloor \nabla &= F  >  grad
\end{align*}
